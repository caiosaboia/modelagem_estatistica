# Criar um modelo para prever Ozone usando as observações não nulas
modelo_imputacao <- lm(Ozone ~ Solar.R + Wind + Temp, data = airquality_data, na.action = na.exclude)

# Prever os valores ausentes de Ozone
ozone_pred <- predict(modelo_imputacao, newdata = airquality_data)
summary(ozone_pred)
